SQL Schema & Sample Data

-- Employees
CREATE TABLE employees (
    employee_id INT PRIMARY KEY,
    name VARCHAR(100)
);

-- Tasks
CREATE TABLE tasks (
    task_id INT PRIMARY KEY,
    title VARCHAR(100),
    assigned_to INT,
    workload_percent INT,
    FOREIGN KEY (assigned_to) REFERENCES employees(employee_id)
);
