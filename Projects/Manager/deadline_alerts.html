SQL Schema & Sample Data

-- Projects
CREATE TABLE projects (
    project_id INT PRIMARY KEY,
    name VARCHAR(100),
    description TEXT
);

-- Tasks
CREATE TABLE tasks (
    task_id INT PRIMARY KEY,
    project_id INT,
    title VARCHAR(100),
    deadline DATE,
    alert_status VARCHAR(20),
    FOREIGN KEY (project_id) REFERENCES projects(project_id)
);
